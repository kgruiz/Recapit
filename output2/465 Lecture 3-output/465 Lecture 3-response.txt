\end{block}

\begin{frame}
\frametitle{Multinomial theorem}

Recall that the multinomial coefficient
\[
\binom{n}{n_1 \cdots n_j} \stackrel{\text{def}}{=} \frac{n!}{n_1! \cdots n_j!}
\]
is the number of words in the alphabet $\{x_1, \dots, x_j\}$ that consist of
$n_1$ copies of letter $x_1$, $\dots$, $n_j$ copies of letter $x_j$. Here $n = n_1 + \dots + n_j$.
\end{frame}

\begin{frame}
\frametitle{Multinomial theorem}
\[
(x_1 + \dots + x_j)^n = \sum_{n_1 + \dots + n_j = n} \binom{n}{n_1 \cdots n_j} x_1^{n_1} \cdots x_j^{n_j}.
\]
\end{frame}

\begin{frame}
\frametitle{Proof}
Expand $(x_1 + \dots + x_j)^n$ as the sum of all words of length $n$ in the
alphabet $\{x_1, \dots, x_j\}$, viewing words as monomials. Collect terms.
\end{frame}

\begin{frame}
\frametitle{Binomial identities via the Binomial Theorem (2)}

\begin{block}{Example}
Start with
\[ (x + 1)^n \cdot (x + 1)^m = (x + 1)^{n+m}. \]
Use the Binomial Theorem to rewrite it as
\[ \sum_{k=0}^n \binom{n}{k} x^k \cdot \sum_{j=0}^m \binom{m}{j} x^j = \sum_{\ell=0}^{n+m} \binom{n+m}{\ell} x^\ell. \]
Take the coefficient of $x^\ell$ on both sides to get
\[ \sum_k \binom{n}{k} \binom{m}{\ell - k} = \binom{n+m}{\ell}. \]
\end{block}

\end{frame}

\begin{block}{Problem}
How many different poker hands can be dealt from a double deck of
$2 \times 52$ cards?
\vspace{0.2cm}

We will solve a more general problem, for hands consisting of $k$ cards.

Set $h_k = \# \{ k\text{-card hands which can be dealt from a double deck} \}$.

\begin{align*}
\sum_k h_k x^k &= (1 + x + x^2)^{52} \\
&= \underbrace{(1 + x + x^2)(1 + x + x^2) \cdots (1 + x + x^2)}_{\text{52 factors}} \text{ [why?]}.
\end{align*}
In other words, $h_k$ is simply the coefficient of $x^k$ in $(1 + x + x^2)^{52}$.
\end{block}

\begin{frame}
\frametitle{Another binomial identity}

\begin{block}{Theorem}
\[
\sum_{k} k \binom{n}{k} = n 2^{n-1}
\]
\end{block}

\begin{block}{Example: $n=5$}
\[
0 \cdot 1 + 1 \cdot 5 + 2 \cdot 10 + 3 \cdot 10 + 4 \cdot 5 + 5 \cdot 1 = 80 = 5 \cdot 2^4.
\]
\end{block}

\begin{block}{Proof}
Suppose we need to choose a \textbf{subset} of an $n$-element set, then pick an element of this subset (e.g., a \textbf{committee} and its chairperson).
The number of such choices can be computed in two different ways:
\begin{itemize}
    \item choose a $k$-element subset, then pick one its $k$ elements;
    \item pick one of the $n$ elements, then add a subset of the rest.
\end{itemize}
\end{block}

\end{frame}

\begin{frame}
\frametitle{Proof of Newton's generalized Binomial Theorem}

Viewing $(x + y)^{\alpha}$ as a function of $x$, we apply Taylor's formula:
\begin{align*}
f(x) &= (x + y)^{\alpha}, \\
f'(x) &= \alpha(x + y)^{\alpha-1}, \\
f''(x) &= \alpha(\alpha - 1)(x + y)^{\alpha-2}, \\
&\dots \\
f^{(k)}(x) &= \alpha(\alpha - 1) \cdots (\alpha - k + 1)(x + y)^{\alpha-k}, \\
&\dots \\
f(x) = (x + y)^{\alpha} &= \sum_{k=0}^{\infty} \frac{f^{(k)}(0)}{k!} x^k \\
&= \sum_{k} \frac{\alpha(\alpha - 1) \cdots (\alpha - k + 1) y^{\alpha-k}}{k!} x^k \\
&= \sum_{k} \binom{\alpha}{k} x^k y^{\alpha-k}.
\end{align*}

\end{frame}

\frametitle{Generalized Binomial Theorem}
\begin{block}{Meta-Theorem [I. Newton]}
The formula
\[
(x+y)^\alpha = \sum_{k=0}^\infty \binom{\alpha}{k} x^k y^{\alpha-k}
\]
holds whenever it makes sense.
\end{block}

In particular, it holds in each of the following cases:
\begin{itemize}
    \item when $\alpha$ is an integer (possibly negative);
    \item when $x+y$ and $y$ are positive real numbers;
    \item in the realm of \textit{formal power series}, to be discussed soon.
\end{itemize}

\end{frame}

\frametitle{Generalized binomial coefficients}
\end{frame}

\begin{frame}
\frametitle{Definition}
\noindent For $\alpha$ a \textit{complex number} and $k$ a nonnegative integer, the \textit{generalized} binomial coefficient $\binom{\alpha}{k}$ is defined by
\[
\binom{\alpha}{k} \stackrel{\text{def}}{=} \frac{\alpha(\alpha - 1) \cdots (\alpha - k + 1)}{k!}.
\]

\noindent When $\alpha = n$ is a nonnegative integer, we recover the usual definition:
\[
\binom{n}{k} = \frac{n(n-1) \cdots (n-k+1)}{k!} = \frac{n!}{(n-k)!k!}.
\]

\noindent Note that for $k > n$, we get $\binom{n}{k} = 0$.
\end{frame}

\frametitle{Generalized binomial coefficients: the case $\alpha = -n$}

The generalized binomial coefficients are defined by
\[
\binom{\alpha}{k} \stackrel{\text{def}}{=} \frac{\alpha(\alpha - 1)\cdots(\alpha - k + 1)}{k!}.

\begin{block}{Example: $\alpha = -n$}
\begin{align*}
\binom{-n}{k} &= \frac{(-n)(-n-1)\cdots(-n - k + 1)}{k!} \\
&= (-1)^k \frac{(n+k-1)\cdots(n+1)n}{k!} \\
&= (-1)^k \binom{n+k-1}{k}.
\end{align*}
\end{block}

\begin{block}{Problem}
What is the largest coefficient of a monomial in the expansion of the polynomial $(x + y + z)^6$?
\end{block}

\begin{block}{Solution}
For $a + b + c = 6$, the largest value of
\[
\binom{6}{a \ b \ c} = \frac{6!}{a! \, b! \, c!}
\]
is attained at $a = b = c = 2$ [why?], so the answer is $\frac{6!}{2^3} = 90$.
\end{block}

\begin{frame}
\frametitle{Generating functions}

\begin{block}{Definition}
Let
\[ h_0, h_1, h_2, h_3, \dots \]
be a finite (resp., infinite) sequence of nonnegative integers.
The \textcolor{blue}{generating function} of this sequence is the polynomial (resp., \textcolor{blue}{formal power series})
\[ h(x) = h_0 + h_1x + h_2x^2 + h_3x^3 + \dots = \sum_k h_kx^k. \]
\end{block}

\begin{block}{Example}
Let $h_k = k+1$, i.e., we look at the sequence $1, 2, 3, \dots$ Then
\[ h(x) = 1 + 2x + 3x^2 + 4x^3 + \dots = \sum_k (k+1)x^k = (1-x)^{-2}. \]
\end{block}

\end{frame}

\begin{frame}
\frametitle{Binomial Theorem for negative powers}

Setting $\alpha = -n$ in Newton's theorem, we get:
\[
(1+x)^{-n} = \sum_{k=0}^{\infty} \binom{-n}{k} x^k = \sum_{k=0}^{\infty} (-1)^k \binom{n+k-1}{k} x^k = \sum_{k=0}^{\infty} (-1)^k \binom{n+k-1}{n-1} x^k.
\]

\vspace{0.5em}
\textbf{Corollary}
\[
(1-x)^{-n} = \sum_{k=0}^{\infty} \binom{n+k-1}{n-1} x^k.
\]

\vspace{0.5em}
\textbf{Examples}
\[
(1-x)^{-1} = \sum_{k} \binom{k}{0} x^k = \sum_{k} x^k,
\]
\[
(1-x)^{-2} = \sum_{k} \binom{k+1}{1} x^k = \sum_{k} (k+1)x^k.
\]

\end{frame}

\begin{frame}
\frametitle{Why generating functions?}

In a typical application, $h_k$ is an answer to an enumerative question
that depends on $k$.

\begin{block}{Example}
(Here $h_k$ = number of $k$-element subsets of an $n$-element set.)
The generating function for the $n$ th row of Pascal's triangle is
\[
\sum_{k} \binom{n}{k} x^k = (1+x)^n.
\]
\end{block}

The generating function is often simpler than its coefficients.

One can use algebraic manipulations with generating functions to
obtain results about sequences.
\end{frame}

\begin{frame}
\frametitle{Generating functions: Example 1}

\begin{block}{Problem}
How many different poker hands can be dealt from a double deck of
$2 \times 52$ cards? (Each poker hand consists of 5 cards.)

The two instances of each card are indistinguishable from each other.
\end{block}

\begin{block}{Solution \#1}
Each of the 52 types of cards will be drawn either 0 or 1 or 2 times.
Each potential drawing can be encoded by a 52-letter word in the
alphabet $\{0, 1, 2\}$.

For a drawing to contain 5 cards, the number of occurences of 0, 1, 2
must be either $(47, 5, 0)$ or $(48, 3, 1)$ or $(49, 1, 2)$. Hence the answer is
\[
\binom{52}{47\ 5\ 0} + \binom{52}{48\ 3\ 1} + \binom{52}{49\ 1\ 2} = 3748160.
\]
\end{block}

\end{frame}
