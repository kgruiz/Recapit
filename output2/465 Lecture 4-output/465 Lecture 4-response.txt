\begin{frame}
\frametitle{Weight functions}

Let $\alpha: A \to \mathbb{Z}_{\geq 0}$ be a ``weight function" on a set $A$.

For each value $k$, let
\[
h_k = \text{number of elements } a \in A \text{ with } \alpha(a) = k.
\]
(We assume that $h_k < \infty$.)

\vspace{0.5em}
\fcolorbox{lightgray}{lightgray}{
\begin{minipage}{\dimexpr\textwidth-2\fboxsep-2\fboxrule\relax}
\textbf{Observation}
\end{minipage}
}

\vspace{0.5em}
\[
h(x) \stackrel{\text{def}}{=} \sum_{k} h_k x^k = \sum_{a \in A} x^{\alpha(a)}.
\]

We will frequently deal with generating functions defined in this way.

If $A$ is a finite set, then the generating function $h(x)$ can be viewed as a ``refinement" of $|A|$, the cardinality of $A$. Indeed, $h(1) = |A|$.

\end{frame}

\begin{frame}
Generating functions: Example 3

The formula below was derived from Newton's Binomial Theorem:

(*)
\begin{equation*}
    (1-x)^{-n} = \sum_{k=0}^\infty \binom{n+k-1}{n-1}x^k.
\end{equation*}

\vspace{0.5cm}
\begin{block}{Alternative proof of (*)}
    We have
    \begin{equation*}
        (1-x)^{-n} = \underbrace{(1+x+x^2+\cdots)(1+x+x^2+\cdots)\cdots(1+x+x^2+\cdots)}_{n} = \sum_k \boxed{\phantom{?}} x^k
    \end{equation*}
    where
    \begin{align*}
        \boxed{\phantom{?}} &= \text{number of nonnegative integer solutions} \\
        & \qquad \text{of the equation } a_1 + \cdots + a_n = k. \\
        &= \text{number of weak compositions of } k \text{ with } n \text{ parts} \\
        &= \binom{n+k-1}{n-1}.
    \end{align*}
\end{block}

\end{frame}

\begin{frame}
\frametitle{Generating functions: Example 2}

\begin{block}{Problem}
How many ways are there to change a dollar into coins?
\end{block}

\begin{block}{Solution}
Let $h_k$ denote the number of solutions of
\[n_1 + 5n_2 + 10n_3 + 25n_4 + 50n_5 + 100n_6 = k\]
in nonnegative integers. We would like to compute $h_{100}$. We have
\begin{align*}
\sum_k h_k x^k &= (1+x+x^2+\cdots)(1+x^5+x^{10}+\cdots)(1+x^{10}+x^{20}+\cdots) \cdots \\
&= \frac{1}{(1 - x)(1 - x^5)(1 - x^{10})(1 - x^{25})(1 - x^{50})(1 - x^{100})}
\end{align*}
The coefficient of $x^{100}$ in the power series expansion of this function can be computed using any symbolic computation software.

\textbf{Answer:} 293.
\end{block}

\end{frame}

\begin{frame}
\frametitle{Additivity for weight functions}

Suppose that we have a bijection of the form
\[
A \leftrightarrow B \times C \times D \times \cdots
\]
\[
a \leftrightarrow (b, c, d, \ldots).
\]
and that
\[
\begin{aligned}
\alpha &: A \to \mathbb{Z}_{\geq 0} \\
\beta &: B \to \mathbb{Z}_{\geq 0} \\
\gamma &: C \to \mathbb{Z}_{\geq 0} \\
\delta &: D \to \mathbb{Z}_{\geq 0} \\
& \cdots
\end{aligned}
\]
are weight functions as above satisfying the \textcolor{blue}{additivity condition}
\[
\alpha(a) = \beta(b) + \gamma(c) + \delta(d) + \cdots.
\]

\end{frame}

\begin{frame}
\frametitle{Multiplication Principle for generating functions}

The Multiplication Principle for generating functions is a refinement of the ordinary Multiplication Principle:

\begin{block}{Theorem}
Let
\begin{align*}
    A &\leftrightarrow B \times C \times D \times \cdots \\
    a &\leftrightarrow (b, c, d, \dots) \\
    \alpha(a) &= \beta(b) + \gamma(c) + \delta(d) + \cdots
\end{align*}
be as above. Then
\[
    \sum_{a \in A} x^{\alpha(a)} = \left( \sum_{b \in B} x^{\beta(b)} \right) \left( \sum_{c \in C} x^{\gamma(c)} \right) \left( \sum_{d \in D} x^{\delta(d)} \right) \cdots
\]

Setting $x = 1$ recovers the ordinary multiplication principle.
\end{block}

\end{frame}

\frametitle{Multiplication Principle for generating functions: examples}

\begin{block}{Changing money into coins (again)}
Let $h_k =$ number of ways to change $k$ cents into coins. We set
\begin{align*}
A &= \{\text{distinct collections of several coins}\}; \\
B &= \{\text{collections of several pennies}\}; \\
C &= \{\text{collections of several nickels}\};
\end{align*}
and so on. Each weight function $\alpha, \beta, \gamma, \dots$ assigns to a collection of coins its value in cents. The additivity condition holds. Therefore
\begin{align*}
\sum_{k} h_k x^k &= (1 + x + x^2 + \dots)(1 + x^5 + x^{10} + \dots)(1 + x^{10} + x^{20} + \dots) \dots \\
&= (1-x)^{-1}(1-x^5)^{-1}(1-x^{10})^{-1} \dots (1-x^{100})^{-1}.
\end{align*}
\end{block}

\begin{frame}
\frametitle{Counting permutations by the number of inversions}

The bijection
\[
S_n \rightarrow \{0\} \times \{0, 1\} \times \{0, 1, 2\} \times \cdots \times \{0, 1, \dots, n-1\}
\]
\[
w \mapsto (c_1, c_2, c_3, \dots)
\]
satisfies the additivity condition if we assume that
\begin{itemize}
    \item the weight of $w$ is $\text{inv}(w)$,
    \item the weight of each entry $c_i$ is $c_i$.
\end{itemize}
Applying the Multiplication Principle for generating functions, we get:

\begin{block}{Theorem}
The generating function for counting permutations in $S_n$ with respect to the number of inversions is given by
\[
\sum_{w \in S_n} x^{\text{inv}(w)} = 1 \cdot (1 + x)(1 + x + x^2) \cdots (1 + \cdots + x^{n-1}).
\]
\end{block}

\end{frame}

\begin{frame}
\frametitle{Code of a permutation}
The \textbf{code} of a permutation $w \in S_n$ is the sequence $(c_1, \ldots, c_n)$ where
\[c_k = \#\{\text{entries to the right of } k \text{ which are } < k\}.\]
In other words: place $1, 2, 3, \ldots$, one by one, and record the number
of inversions each new entry creates:

\begin{center}
\begin{tabular}{|c|c|c|}
\hline
$w$ & $\text{inv}(w)$ & $\text{code}(w)$ \\
\hline
123 & 0 & (0, 0, 0) \\
132 & 1 & (0, 0, 1) \\
213 & 1 & (0, 1, 0) \\
231 & 2 & (0, 1, 1) \\
312 & 2 & (0, 0, 2) \\
321 & 3 & (0, 1, 2) \\
\hline
\end{tabular}
\end{center}

\bigskip
\noindent
\textbf{Observations}
\begin{itemize}
    \item $\text{inv}(w) = c_1 + \cdots + c_n$;
    \item each $c_k$ takes values $0, 1, \ldots, k-1$.
\end{itemize}

\end{frame}

\begin{frame}
\frametitle{Counting permutations by the number of cycles}

\begin{block}{Theorem}
The generating function for counting permutations in $S_n$ with respect to the number of cycles is given by
\[
\sum_{w \in S_n} x^{c(w)} = x(x + 1)(x + 2)\cdots (x + n - 1).
\]
\end{block}

\begin{block}{Proof}
Each permutation $w \in S_n$ is encoded by a sequence $(b_1, \dots, b_n)$.
The additivity condition is satisfied:
\[
c(w) = \beta(b_1) + \cdots + \beta(b_n).
\]
The generating function for each $b_k$ is $x + k - 1$ [why?].

By the Multiplication Principle for generating functions, we obtain
\[
\sum_{w \in S_n} x^{c(w)} = \prod_{k} (x + k - 1).
\]
\end{block}

\end{frame}
